  , Data.Set.fromAscList
#if MIN_VERSION_containers(0,5,8)
  , Data.Set.fromDescList
#endif
  , Data.Set.fromDistinctAscList
#if MIN_VERSION_containers(0,5,8)
  , Data.Set.fromDistinctDescList
#endif
  ) where

import qualified Data.Set
