-- | Semigroup action.  It should satisfy:
--
-- @
--   x <| s0 <| s1 = x <| s0 <> s1
-- @
class Semigroup s => SAct s x where
  (<|) :: x -> s -> x

-- binds less than '<>'
infixr 5 <|
