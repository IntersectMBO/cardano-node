platform Reg
reg
 = case Reg
reg of
   RegVirtual VirtualReg
vr -> VirtualReg -> RegClass
classOfVirtualReg VirtualReg
vr
   RegReal RealReg
rr -> Platform -> RealReg -> RegClass
targetClassOfRealReg Platform
platform RealReg
