narrowU :: Width -> Integer -> Integer Source #

Narrow a signed or unsigned value to the given width. The result will reside in [0, +2^width).

>>> narrowU W8 256    == 256
>>> narrowU W8 255    == 255
>>> narrowU W8 128    == 128
>>> narrowU W8 127    == 127
>>> narrowU W8 0      == 0
>>> narrowU W8 (-127) == 129
>>> narrowU W8 (-128) == 128
>>> narrowU W8 (-129) == 127
>>> narrowU W8 (-255) == 1
>>> narrowU W8 (-256) == 0

narrowS :: Width -> Integer -> Integer Source #

Narrow a signed value to the given width. The result will reside in [-2^(width-1), +2^(width-1)).

>>> narrowS W8 256    == 0
>>> narrowS W8 255    == -1
>>> narrowS W8 128    == -128
>>> narrowS W8 127    == 127
>>> narrowS W8 0      == 0
>>> narrowS W8 (-127) == -127
>>> narrowS W8 (-128) == -128
>>> narrowS W8 (-129) == 127
>>> narrowS W8 (-255) == 1
>>> narrowS W8 (-256) == 0
